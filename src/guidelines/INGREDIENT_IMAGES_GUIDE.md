```tsx

  />
  <h3>Vitamin C</h3>
</div>
```

### **Option 2: Pre-fetch All Images**

Create a script to fetch all images at build time:

```tsx
import { ingredientImageConfigs } from './data/ingredientImages';

async function fetchAllIngredientImages() {
  const imageMap = {};
  
  for (const config of ingredientImageConfigs) {
    const imageUrl = await unsplash_tool({ 
      query: config.primarySearch 
    });
    
    imageMap[config.ingredientName] = imageUrl;
  }
  
  return imageMap;
}
```

### **Option 3: Use Data File Directly**

Import the configurations and use them in your ingredient cards:

```tsx
import { getIngredientImageConfig } from './data/ingredientImages';

function IngredientCard({ ingredient }) {
  const imageConfig = getIngredientImageConfig(ingredient.name);
  
  return (
    <div className="ingredient-card">
      {/* Use imageConfig.primarySearch with unsplash_tool */}
      <p className="text-sm">{imageConfig.notes}</p>
    </div>
  );
}
```
